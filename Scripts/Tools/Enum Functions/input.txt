enum class Conditional
{
	/// <summary>
	/// ==
	/// </summary>
	Equal,

	/// <summary>
	/// !=
	/// </summary>
	NotEqual,

	/// <summary>
	/// >
	/// </summary>
	GreaterThan,

	/// <summary>
	/// >=
	/// </summary>
	GreaterThanOrEqualTo,

	/// <summary>
	/// <
	/// </summary>
	LessThan,

	/// <summary>
	/// <=
	/// </summary>
	LessThanOrEqualTo,
};
