/// <summary>
/// Determines how the image data is filtered.
/// </summary>
enum class Filter
{
	/// <summary>
	/// Invalid.
	/// </summary>
	Undefined = 0,

	/// <summary>
	/// Use the nearest pixel.
	/// </summary>
	Nearest = 1,

	/// <summary>
	/// Linearly interpolate between pixels.
	/// </summary>
	Linear = 2,
};
