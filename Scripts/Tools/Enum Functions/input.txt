	/// <summary>
	/// Determines how the polygons are being drawn.
	/// </summary>
	enum class ShaderPolygonMode
	{
		/// <summary>
		/// Invalid.
		/// </summary>
		Undefined = 0,

		/// <summary>
		/// Fill the polygon.
		/// </summary>
		Fill = 1,

		/// <summary>
		/// Draw the outlines of the polygon.
		/// </summary>
		Line = 2,

		/// <summary>
		/// Draw the points of the polygon.
		/// </summary>
		Point = 3,
	};
