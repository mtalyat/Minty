/// <summary>
/// Determines how the image is used.
/// </summary>
enum class ImageUsage
{
	/// <summary>
	/// Invalid.
	/// </summary>
	Undefined = 0,

	/// <summary>
	/// The image is sampled.
	/// </summary>
	Sampled = 1,

	/// <summary>
	/// The image is used for storage.
	/// </summary>
	Storage = 2,

	/// <summary>
	/// The image is used for input attachment.
	/// </summary>
	Color = 4,

	/// <summary>
	/// The image is used for depth/stencil.
	/// </summary>
	DepthStencil = 8
};
