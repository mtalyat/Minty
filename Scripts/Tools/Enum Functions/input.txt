	/// <summary>
	/// Determines what data is being inputted to the shader.
	/// </summary>
	enum class ShaderInputRate
	{
		/// <summary>
		/// Invalid.
		/// </summary>
		Undefined = 0,

		/// <summary>
		/// The shader is using vertex data.
		/// </summary>
		Vertex = 1,

		/// <summary>
		/// The shader is using instance data.
		/// </summary>
		Instance = 2
	};
