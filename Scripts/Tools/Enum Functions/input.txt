/// <summary>
/// Determines how the image is used in the pipeline.
/// </summary>
enum class ImageAspect
{
	/// <summary>
	/// Invalid.
	/// </summary>
	Undefined = 0,

	/// <summary>
	/// Used for color.
	/// </summary>
	Color = 1,

	/// <summary>
	/// Used for depth.
	/// </summary>
	Depth = 2,
};
